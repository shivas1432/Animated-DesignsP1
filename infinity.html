const createInfinityAnimation = () => { const container = document.getElementById('infinity-container'); const dotCount = 48; const animationDuration = 3800; for (let i = 0; i < dotCount; i++) { const dot = document.createElement('div'); dot.classList.add('dot'); container.appendChild(dot); const delay = (i / dotCount) * animationDuration; dot.style.animation = `fadeInOut ${animationDuration}ms infinite ${delay}ms`; updateDotPosition(dot, i, dotCount); } }; const updateDotPosition = (dot, index, total) => { const t = (index / total) * Math.PI * 2; const width = 360; const height = 120; const x = width / 2 * (Math.sin(t) / (1 + Math.pow(Math.cos(t), 2))); const y = height / 2 * (Math.sin(t) * Math.cos(t) / (1 + Math.pow(Math.cos(t), 2))); dot.style.left = (width / 2 + x) + 'px'; dot.style.top = (height / 2 + y) + 'px'; const size = 9 + Math.sin(t * 2) * 5; dot.style.width = size + 'px'; dot.style.height = size + 'px'; dot.style.filter = 'none'; }; createInfinityAnimation();
