/* Button Click Effects */
function playSuccessAnimation(button) {
  const card = button.closest('.card');
  card.classList.add('success-effect');
  
  const confettiContainer = document.createElement('div');
  confettiContainer.classList.add('confetti-container');
  card.appendChild(confettiContainer);
  
  const greenRing = document.createElement('div');
  greenRing.classList.add('green-ring');
  confettiContainer.appendChild(greenRing);
  
  for (let i = 0; i < 50; i++) {
    const confetti = document.createElement('div');
    confetti.classList.add('confetti');
    confetti.style.backgroundColor = getRandomColor();
    confetti.style.width = Math.random() * 8 + 4 + 'px';
    confetti.style.height = Math.random() * 4 + 2 + 'px';
    confetti.style.left = '50%';
    confetti.style.top = '50%';
    
    const x = (Math.random() - 0.5) * 150;
    const y = (Math.random() - 0.5) * 150;
    const r = Math.random() * 360;
 
  }
  
  setTimeout(() => {
    card.classList.remove('success-effect');
    confettiContainer.remove();
  }, 1500);
}
